linkP5 instance mode

Use the a, s, w and z keys to move the quadrille:

The markdown of the above sketch looks like this:

1link> :P5 lib1=https://cdn.jsdelivr.net/gh/objetos/p5.quadrille.js/p5.quadrille.min.js, sketch=/docs/sketches/colors.js, width=400, height=400, version=1.1.8

and the colors.js p5 instance mode like this:

colors.js
1linkvar myp5 = new p5((p) => {

2link  const ROWS = 20;

3link  const COLS = 20;

4link  const LENGTH = 20;

5link  var quadrille;

6link  var clone;

7link  var x = 2, y = 2;

8link

9link  p.setup = function () {

10link    p.createCanvas(COLS * LENGTH, ROWS * LENGTH);

11link    quadrille = p.createQuadrille([[p.color('cyan'), '游놓',               0    ],

12link                                   [0,               '游뱂',              '游뗻' ],

13link                                   [0,               p.color('#770811'), 0   ],

14link                                   ['g',             'o',                'l' ]

15link                                  ]);

16link    clone = quadrille.clone();

17link    clone.reflect();

18link  };

19link

20link  p.draw = function () {

21link    p.background('#007ACC');

22link    p.drawQuadrille(quadrille, x, y, LENGTH, 2, 'green');

23link    p.drawQuadrille(clone, 12, 2, LENGTH, 0);

24link  };

25link

26link  p.keyPressed = function () {

27link    if (p.keyCode === p.LEFT_ARROW) {

28link      quadrille.reflect();

29link    } else if (p.keyCode === p.RIGHT_ARROW) {

30link      quadrille.rotate();

31link    }

32link    if (p.key === 'a') {

33link      x = x > 0 ? x - 1 : x;

34link    }

35link    if (p.key === 's') {

36link      x = x < COLS - quadrille.width ? x + 1 : x;

37link    }

38link    if (p.key === 'w') {

39link      y = y > 0 ? y - 1 : y;

40link    }

41link    if (p.key === 'z') {

42link      y = y < ROWS - quadrille.height ? y + 1 : y;

43link    }

44link  };

45link}, "colors"); // --> the id should be the same file name
